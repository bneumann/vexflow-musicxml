export class XmlSerializer {
  /**
   * @constructor
   * @param {string} xDoc - The string representation of an XML document
   */
  constructor(xDoc) {
    const domParser = new DOMParser();
    this.xDoc = domParser.parseFromString(xDoc, 'text/xml');
  }
}
